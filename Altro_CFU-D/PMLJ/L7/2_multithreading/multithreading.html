function mthr_sum1(A)
    r = 0.0

Key takeaways from mthr_sumX:

  • Beware shared state across threads
    • Protect yourself by using atomics (or locks/mutexes)
    • Better: separate states manually in @threads for i in 1:nthreads()
      • Simpler: use array, with separate entries per thread

Recall mthr_sum4

function mthr_sum4(A)
    R = zeros(eltype(A), nthreads())
    @threads for i in eachindex(A)
        @inbounds R[threadid()] += A[i]
    end
    r = 0.0
    for i in eachindex(R)
        @inbounds r += R[i]
    end
    return r
end

function serialpi(n)
    inside = 0
    for i in 1:n
        x, y = rand(), rand()
        inside += (x^2 + y^2 <= 1)
    end
    return 4 * inside / n
end

Naive multithreaded version

function threadedpi(n)
    inside = zeros(Int, nthreads())
    @threads for i in 1:n
        x, y = rand(), rand()
        @inbounds inside[threadid()] += (x^2 + y^2 <= 1)
    end
    return 4 * sum(inside) / n
end

Solution: use separate RNGs

const ThreadRNG = Vector{Random.MersenneTwister}(undef, nthreads())
@threads for i in 1:nthreads()
    ThreadRNG[Threads.threadid()] = Random.MersenneTwister()
end

function threadedpi2(n)
    inside = zeros(Int, nthreads())
    len, rem = divrem(n, nthreads())
    rem == 0 || error("use a multiple of $(nthreads()), please!")
    @threads for i in 1:nthreads()
        rng = ThreadRNG[threadid()]
        v = 0
        for j in 1:len
            x, y = rand(rng), rand(rng)
            v += (x^2 + y^2 <= 1)
        end
        inside[threadid()] = v
    end
    return 4 * sum(inside) / n
end

Multithreaded BLAS

using LinearAlgebra
BLAS.set_num_threads(nthreads())

function threaded_matmul(As)
  first_idxs = zeros(length(As))
  @threads for i in eachindex(As)
    @inbounds first_idxs[i] = (As[i]'*As[i])[1]
  end
  first_idxs
end

Beware of false sharing

function mthr_sum5(x)
    partsum = zeros(eltype(x), nthreads())
    @threads for i in eachindex(x)
        @inbounds partsum[threadid()] += x[i]
    end
    return sum(partsum)
end

However…

function mthr_sum6(x)
  gap = 8  # Adjust based on cache line size
  partsum = zeros(eltype(x), nthreads()*gap)
  @threads for i in eachindex(x)
    idx = (threadid()-1)*gap + 1
    @inbounds partsum[idx] += x[i]
  end
  return sum(partsum)
end

@btime mthr_sum6($A)

function mthr_sum7(x)
  nth = nthreads()
  partsum = zeros(eltype(x), nth)
  @threads for t in 1:nth
    local_sum = zero(eltype(x))
    for i in t:nth:length(x)
      local_sum += x[i]
    end
    partsum[t] = local_sum
  end
  return sum(partsum)
end

@btime mthr_sum7($A)
  5.955 ms (24 allocations: 2.31 KiB)
4.999367844691325e6

Keep in mind

  • Multithreading: start Julia with JULIA_NUM_THREADS and tack a @threads on loop
  • Beware
    • to set JULIA_NUM_THREADS appropiately
    • of shared state (for performance and correctness)
    • of global state (can be far from obvious)
    • of false sharing (if Julia/LLVM don’t handle it)
  • Think carefully about how to design parallel algorithms
